JQuery Tutorials : Gráficos de Barras

Nesta página é usado o plugin jQuery "jqBarGraph"

jqBarGraph é um plugin jQuery que lhe dá liberdade para mostrar seus dados como gráficos. Existem três tipos de gráficos: simples, multiplos e em pilha.
Apenas é necessário passar os dados para o plugin.

Necessário fazer download do "jqBarGraph". Para o código seguinte funcionar foi necessário: criar as sub-directorias jqBarGraph\js (para onde foi copiado o ficheiro "jqBarGraph.1.1.min.js" do download)

Gráfico Barras

CSS
        <style>
          #grafico1
           {
            width: 600px;
            height: 300px;
            margin: 30px auto;
            font-family: sans-serif;
            font-size: 14px;
            font-weight: bold;
            color: rgb(100, 100, 100);
          } 
        </style>
HTML
          <div id="grafico1"></div>
JAVASCRIPT
          <script src="external/jquery/jquery.js"></script>
          <script src="jquery-ui.js"></script>
          <script src="jqBarGraph/js/jqBarGraph.1.1.min.js"></script>
      
          <script>
          $(function () {
            var DadosGrafico1 = new Array([10, "Jan", "#00429d"], [15, "Fev", "#3965ad"], [12, "Mar", "#5889bd"], [13, "Abr", "#74aece"], [9, "Mai", "#91d4df"], [15, "Jun", "#c3f7f6"]);
            $("#grafico1").jqBarGraph({
              data: DadosGrafico1,
              width: 600,
              height: 300,
              barSpace: 5,
              speed: 1,
              //sort:"asc",
              //sort:"desc",
              prefix: "EUR ",
              postfix: ",00",
              position: "bottom", //default
              //position:"top",
            });
          });
          </script>

Gráfico Barras Compostas

CSS
        <style>
          #grafico2
           {
            width: 600px;
            height: 300px;
            margin: 30px auto;
            font-family: sans-serif;
            font-size: 14px;
            font-weight: bold;
            color: rgb(100, 100, 100);
          } 
        </style>
HTML
          <div id="grafico2"></div>
JAVASCRIPT
          <script src="external/jquery/jquery.js"></script>
          <script src="jquery-ui.js"></script>
          <script src="jqBarGraph/js/jqBarGraph.1.1.min.js"></script>
      
          <script>
          $(function () {
            var DadosGrafico2 = new Array([[18, 50, 51], "2019"], [[20, 76, 64], "2020"], [[15, 81, 73], "2021"], [[25, 86, 54], "2022"]);
            $("#grafico2").jqBarGraph({
              data: DadosGrafico2,
              width: 600,
              height: 300,
              barSpace: 5,
              speed: 1,
              colors: ["#00429d", "#669bc6", "#91d4df"],
              legend:true,
              legends:['Operação','Produção','Distrib.']
            });
          });
          </script>

            font-size: 14px;

HTML
          <div id="grafico3"></div>
JAVASCRIPT
          <script src="external/jquery/jquery.js"></script>

          <script src="jqBarGraph/js/jqBarGraph.1.1.min.js"></script>
      
          <script>
          $(function () {
            var DadosGrafico3 = new Array([[18, 50, 51], "2019"], [[20, 76, 64], "2020"], [[15, 81, 73], "2021"], [[25, 86, 54], "2022"]);

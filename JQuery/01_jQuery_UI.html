Datepicker

Campos de data com DatePicker associado, com meses, dias de semana e botões traduzidos para português.

Formato de data: "dd/mm/YYYY".

Data de início tem de ser igual ou superior à data actual.

Data de Fim tem de ser igual ou superior à Data de Início.

Data Inicial:

Data Final:

HTML
          <p>
            Data Inicial:
            <input type="text" id="inicio" />
          </p>
    
          <p>
            Data Final:
            <input type="text" id="fim" />
          </p>
JAVASCRIPT
        <script>
          $(function () {
            var months = [
              "Janeiro",
              "Fevereiro",
              "Março",
              "Abril",
              "Maio",
              "Junho",
              "Julho",
              "Agosto",
              "Setembro",
              "Outubro",
              "Novembro",
              "Dezembro",
            ];
  
            var dayNames = ["Dom", "Seg", "Ter", "Qua", "Qui", "Sex", "Sáb"];
  
            var configini = {
              monthNames: months,
              dayNamesMin: dayNames,
              prevText: "Anterior",
              nextText: "Próximo",
              dateFormat: "dd/mm/yy",
              minDate: "+0",
              maxDate: "+1y+1m",
              onSelect: function (dateText, inst) {
                $("#fim").datepicker(
                  "option",
                  "minDate",
                  $("#inicio").datepicker("getDate")
                );
              },
            };
  
            var configfim = {
              monthNames: months,
              dayNamesMin: dayNames,
              prevText: "Anterior",
              nextText: "Próximo",
              dateFormat: "dd/mm/yy",
              minDate: "+0",
              maxDate: "+1y+1m",
            };
  
            $("#inicio").datepicker(configini);
            $("#fim").datepicker(configfim);
          });
  
        </script>

Slider

Intervalo: 1 a 99, salto de 0.01, Valor Inicial: 50. Actualiza control text.
CSS
        <style>
          .show_value {
            margin-top: 10px;
            border: none;
            outline: none;
            display: block;
            font-size: 36px;
          }
        </style>
HTML
          <div id="slider" name="slider"></div>
          <input
            class="show_value"
            readonly="readonly"
            type="text"
            id="slider_value"
          />
JAVASCRIPT
        <script>
        $(function () {
          $("#slider").slider({
            // range: true,
            // values: [40, 60],
            step: 0.01,
            value: 50,
            min: 1,
            max: 99,
            slide: function (event, ui) {
              $("#slider_value").val(ui.value + " EUR");
            },
          });
          $("#slider_value").val($("#slider").slider("value") + " EUR");
        });
        </script>

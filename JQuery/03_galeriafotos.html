JAVASCRIPT
        <script>
          $(function () {
            //adicionar div legenda
            $('
').insertBefore("ul.album");
            //adicionar div foto
            $('
').insertBefore("#legenda");
            //adicionar tag de imagem
            $("#mainfoto").prepend("");
            //adicionar foto a tag img
            $("#mainfoto img").attr("src", $("#galeria a").attr("href"));
            //adiciona texto a legenda
            $("#legenda").text($("#galeria a img").attr("title"));
  
            function abrirFoto(e) {
              e.preventDefault();
              $("#mainfoto img").attr("src", $(this).parent().attr("href"));
              $("#legenda").text($(this).attr("title"));
            }
  
            $(".thumb").on("click", abrirFoto);
          });
        </script>

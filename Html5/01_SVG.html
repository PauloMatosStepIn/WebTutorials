        <svg xmlns="http://www.w3.org/2000/svg" version="1.1" width="490" height="370" >
          <text x="245" y="50" font-size="24" text-anchor="middle" fill="black">Exemplo Gráfico SVG</text>
          <text x="91" y="343" font-size="12" text-anchor="middle" fill="black">JAN</text>
          <text x="170" y="343" font-size="12" text-anchor="middle" fill="black">FEV</text>
          <text x="250" y="343" font-size="12" text-anchor="middle" fill="black">MAR</text>
            <line x1="91" y1="323" x2="91" y2="150" style="stroke:red;stroke-width:50"></line>
            <line x1="170" y1="323" x2="170" y2="110" style="stroke:blue;stroke-width:50"></line>
            <line x1="250" y1="323" x2="250" y2="80" style="stroke:green;stroke-width:50"></line>
        </svg>
      </div>
CSS
      <style>
      .grafico {
        width:500px;margin:0 auto
      }
    
      .grafico svg {
        background-image:url(imagens/background_grafico.png);
        background-repeat:no-repeat;
      }	
      </style>

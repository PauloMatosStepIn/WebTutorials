HTML
      <div style="text-align: center">
        <audio id="player">
          <source src="audio/Symphony No.6 (1st movement).mp3" type="audio/mpeg" />
          <source src="audio/Symphony No.6 (1st movement).m4a" type="audio/mpeg" />
          <source src="audio/Symphony No.6 (1st movement).ogg" type="audio/ogg" />
        </audio>
  
        <form style="padding: 20px;">
          <input type="button" value="Play" onclick="play()">
          <input type="button" value="Pause" onclick="pause()">
          <input type="button" value="Stop" onclick="stop()">
        </form>
      </div>
JAVASCRIPT
      <script>
      var audio
      function inicial() {
         audio = document.querySelector("#player")
      }
      function play() {
        audio.play()
      }
      function pause() {
        audio.pause()      
      }
      function stop() {
        audio.pause()
        audio.currentTime = 0
      }
      window.onload= inicial  
      </script>

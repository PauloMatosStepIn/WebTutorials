CSS
      <style>
      #janela {
        height: 500px;
        width: 860px;
        margin-top: 30px;
        margin-right: auto;
        margin-left: auto;
      }

      .janela_base {
        position: relative;
        float: left;
        width: 256px;
        height: 256px;
        border: solid 5px #fff;
        background-color: #1a1a1a;
        margin: 0 20px 50px 0;
        box-shadow: 2px 2px 2px #333;
      }

      .janela_foto {
        position: absolute;
        top: 0;
        left: 0;
        width: 256px;
        height: 256px;
        z-index: 1;
        /* importante para esconder a parte que fica de fora */
        /* clip is deprecated, use clip-path */
        clip: rect(0 256px 256px 0);
        clip-path: inset(0);
      }

      .janela_texto {
        position: absolute;
        top: 230px;
        left: 0;
        width: 246px;
        height: 256px;
        /* background-color: #000; */
        background-color: rgba(0, 0, 0, 0.6);
        z-index: 2;
        padding: 5px;
        /* alterar apenas a opacidade em background-color usando rgba()*/
        /* opacity: 0.7; */
        cursor: pointer;

        transition: all 0.4s ease-out;
      }

      .janela_texto:hover {
        top: 100px;
      }

      h3 {
        font-family: 'PT Mono', sans-serif;
        font-size: 18px;
        color: #ffffff;
        letter-spacing: 0em;
        margin: 0;
      }

      div p {
        font-family: 'Trebuchet MS', Arial, Helvetica, sans-serif;
        font-size: 14px;
        color: #fff;
        line-height: 150%;
      }

      .azul {
        color: #008b9c;
      }
      </style>

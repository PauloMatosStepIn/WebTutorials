Come dine with us tonight

HTML
        <div style="padding: 0px" id="janela5b">
          janela5b
          <article>
            <h2>Come dine with us tonight</h2>
          </article>
        </div>
CSS
          <style>
          #janela5b {
            margin: 10px auto 50px;
            width: 700px;
            height: 250px;
            background-color: #fff;
          }
          #janela5b {
            position: relative;
          }
          
          #janela5b:before {
            position: absolute;
            /* needed to display */
            content: '';
            box-shadow: 0px 10px 10px 0px #777;
            /* top ao centro */
            top: 50%;
            /* encostado ao fundo */
            bottom: 0;
            right: 10px;
            left: 10px;
            /* importante para a curvatura ficar mais alongada na horizontal */
            border-radius: 130px/15px;
          }
    
          #janela5b:before {
            z-index: -1;
          }
    
          #janela5b {
            border-radius: 15px/500px;
          }

          article h2 {
            font-family: Georgia, 'Times New Roman', Times, serif;
            font-style: italic;
            text-shadow: 5px 5px 5px #999;
            color: rgb(92, 92, 92);
            margin: 3px 0;
  
            font-size: 36px;
  
            padding-top: 40px;
            line-height: 180%;
          }
          </style>

JAVASCRIPT
  <script>
    $('a[data-bs-toggle="tab"]').on("shown.bs.tab", function (e) {
      // save current clicked tab in local storage
      localStorage.setItem("lastActiveTab", $(this).attr("href"));

      console.log($(this).attr("href"));
    });
    //get last active tab from local storage
    var lastTab = localStorage.getItem("lastActiveTab");
    console.log(lastTab + " tab was last active");

    $(".nav-link").removeClass("active");
    $(".nav-pane").removeClass("show").removeClass("active");

    if (lastTab) {
      $(`button[data-bs-target="${lastTab}"]`).addClass("active");
      $('[href="' + lastTab + '"]')
        .addClass("show")
        .addClass("active");
    } else {
      //default option, if local storage empty
      $('button[data-bs-target="#home"]').addClass("active");
      $('[href="#home"]').addClass("show").addClass("active");
    }
  </script>

Detectar Largura da Janela do Browser w/JavaScript

JavaScript:

  <script>
    function getResolucao() {
      console.log('window.innerWidth: ' + window.innerWidth)
    }
    window.onload = window.onresize = getResolucao

    var largura

    var fileCSS = document.querySelector('#PageCSS')

    /* para funcionar em todos os tipos de dispositivos */
    if (window.innerWidth) {
      largura = window.innerWidth
    } else if (document.documentElement && document.documentElement.clientWidth) {
      largura = document.documentElement.clientWidth
    } else if (document.body) {
      largura = document.body
    }

    if (largura < 415) {
      fileCSS.setAttribute('href', '04_js/phone.css')
    } else {
      fileCSS.setAttribute('href', '04_js/desktop.css')
    }
  </script>

Detectar Posição do Mouse w/JavaScript

JavaScript:

        <script>
          function getMousePosition(e) {
            var posX = document.querySelector('#posicaoX')
            var posY = document.querySelector('#posicaoY')
            posX.innerHTML = e.clientX
            posY.innerHTML = e.clientY
          }

          document.addEventListener('mousemove', getMousePosition)
        </script>

Html:

        Posição do Mouse: <span id="posicaoX"></span> x <span id="posicaoY"></span>

Output:

        Posição do Mouse:  x

Detectar Tecla Pressionada w/JavaScript

JavaScript:

        <script>
        function getTecla(e) {
          /* keyCode = ASCII Code */
          /* key = character */
          var teclaCode = document.querySelector('#keyCode')
          var teclaChar = document.querySelector('#key')
          teclaCode.innerHTML = e.keyCode
          teclaChar.innerHTML = e.key
        }
  
        document.addEventListener('keydown', getTecla)
          </script>

Html:

    Tecla Pressionada (ASCII CODE): <span id="keyCode"></span> Character: <span id="key"></span>

Output:

        Tecla Pressionada (ASCII CODE):  Character:

Detectar Posição de Scroll w/JavaScript

JavaScript:

        <script>
          document.addEventListener('scroll', () => {
            scrollPosValue.innerHTML = window.scrollY
          })
        </script>

Abrir DIV por Posição de Scroll w/JavaScript

JavaScript:

        <script>
          document.addEventListener('scroll', () => {
            if (window.scrollY > 300) {
              AvisoScroll.style.display = 'block'
            } else {
              AvisoScroll.style.display = 'none'
            }
          })
        </script>

Css:

        <style>
          #AvisoScroll {
            position: fixed;
            top: 200px;
            left: 50%;
            width: 300px;
            height: 150px;
            margin-left: -150px;
            background-color: lightgray;
            display: none;
            border: 1px solid darkgray;
            border-radius: 5px;
            box-shadow: 0 0 20px black;
          }
        </style>

Html:

    <div id="AvisoScroll"><p style="padding: 20px">A posição de Scroll Vertical é superior a 300 !</p></div>

A posição de Scroll Vertical é superior a 300 !

Output:

        Caixa no centro da janela quando posição de scroll vertical superior a 300.

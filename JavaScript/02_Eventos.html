Web Javascript Tutorials : Eventos

Eventos de Browser

Load

Quando os recursos da página são carregados

Error

Quando ocorre erro no carregamento

Resize

Quando o Viewport é redimensionado

Scroll

Quando a barra de Scroll é movida

Eventos de DOM (Document Object Model)

Focus

Quando um elemento recebe o Focus

Blur

Quando um elemento perde o Focus

Submit

Quando um Form é submetido

Mouse Event

Quando ocorre um click, mouseover, drag, drop

Media Events

Controle de Video e Audio

CSS Transition Event

Quando uma transação inicializa, termina ou está em execução

Progress Event

Quando o carregamento de um recurso é despoletado

Faça a seu Registo

HTML
            <div class="apresentacao">
              <h2>Faça a seu Registo</h2>
              <form action="02_Eventos.html" method="post">
                  <input type="text"  id="nome"   name="nome"     placeholder="Nome"  />	
                  <input type="email" id="email"  name="email"    placeholder="Email" />	
                  <input type="submit" value="Confirmar Registo"/>  
                  <select id="genero">
                    <option>Indicar género</option>
                    <option value="male">Masculino</option>
                    <option value="female">Feminino</option>
                  </select>	           
              </form>
            </div>
JAVASCRIPT
      <script>
      window.onload = function () {
        var campo = document.querySelector('input#nome')
        campo.onfocus = function(){
          console.log("on focus no campo nome")
        }
        campo.onblur = function(){
          console.log("on blur no campo nome")
        }
        var sexo = document.querySelector('#genero')
        sexo.onchange = function(){
          console.log(sexo.value)
        }
      }    
      </script>

Event Listener

JAVASCRIPT (opção sem parametros):
      <script>
        function FillText(){
          nomeEL.value = "texto from js!"
        }
        function ResetText(){
          FilleEL.value = null
        }
        function iniciar(){
          botaoPreenche.addEventListener("click",FillText)
          botaoLimpa.addEventListener("click",ResetText)
        }
        window.onload = iniciar
      </scFillt>
JAVASCRIPT (opção com parametros):
      <script>
        function SetText(texto){
          nomeEL.value = texto
        }
        function iniciar(){
          botaoPreenche.addEventListener("click",()=>{SetText("texto from js!") })
          botaoLimpa.addEventListener("click",()=>{SetText(null) })
        }
        window.onload = iniciar
      </scFillt>

      <div class="apresentacao">

      </div>

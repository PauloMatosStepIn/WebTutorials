Web Javascript Tutorials : Change Style w/JS
CSS
  <style>
    .azul {
      background-color: blue;
    }
  </style>
JAVASCRIPT
  <script>
    function PreencheCor(cor) {
      janela.style.backgroundColor = cor
    }

    function AdicionaClass(classe) {
      janela.className = classe
    }

    function iniciar() {
      btnAmarelo.addEventListener("click", () => {
        PreencheCor("yellow")
      })
      btnVermelho.addEventListener("click", () => {
        PreencheCor("red")
      })
      
      AdicionaClass("azul")
    }

    window.onload = iniciar
  </script>
HTML
  <main>
    <div id="janela"></div>
    <input type="button" id="btnAmarelo" value="Preenche Amarelo">
    <input type="button" id="btnVermelho" value="Preenche Vermelho">
  </main>

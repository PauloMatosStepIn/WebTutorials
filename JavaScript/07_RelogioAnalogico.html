JAVASCRIPT
      <script>
      var agora

      function mostraHora() {
        agora = new Date()
        var hr = agora.getHours()
        var min = agora.getMinutes()
        var seg = agora.getSeconds()
        /* para dar efeito de continuidade */
        var mil = agora.getMilliseconds()

        console.log(hr + ':' + min + ':' + seg)

        var ponteiro_hr = hr * 30 + min / 2 + (seg * 6) / 360
        var ponteiro_min = min * 6 + (seg * 6) / 60
        var ponteiro_seg = seg * 6 + mil / (1000 / 6)

        document.querySelector('#hora').style.transform = 'rotate(' + ponteiro_hr + 'deg)'
        document.querySelector('#minuto').style.transform = 'rotate(' + ponteiro_min + 'deg)'
        document.querySelector('#segundo').style.transform = 'rotate(' + ponteiro_seg + 'deg)'
      }
      /* Para dar o efeito de continuidade basta actualizar de 50 em 50 milisegundos */
      var intervalo = setInterval(mostraHora, 50)
      </script>

Métodos JS de criação de elemento DOM / HTML usados nesta página
createElement / appendChild (js/js3_AdicionarElemento.js)

JAVASCRIPT
            var novaPizza = document.createElement('li')
            elemento.appendChild(novaPizza)
            novaPizza.innerHTML = 'Nova Pizza'

or (sem definir novaPizza):

JAVASCRIPT
            elemento.appendChild(document.createElement('li')).innerHTML = 'Mais uma Pizza'

or (sem definir elemento ):

JAVASCRIPT
            document.querySelector('#opcoesPizza').appendChild(document.createElement('li')).innerHTML = 'Outra Pizza'

Métodos Set/Get p/Atributos de elemento DOM / HTML usados nesta página
setAttribute / getAttribute (js/js4_SetGetAttributes.js)

JAVASCRIPT
              var imagem = document.querySelector('img')

              imagem.setAttribute('src', 'imagens/outra_pizza.jpg')
              imagem.setAttribute('width', '300px')
              imagem.setAttribute('alt', 'Foto de Outra Pizza')
              
              console.log(imagem.getAttribute('alt'))
